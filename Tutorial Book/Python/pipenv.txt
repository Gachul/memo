# pipenv 설치
# cd [가상환경 설치할 프로젝트 이동]
# pipenv --python [version]
pipenv --python 3.9

# pipenv 삭제
pipenv --rm

# 가상환경 버젼 확인
pipenv run python --version

# 가상환경 초기화
pipenv install >> pipfile, pipfile.lock 생성

# pipfile, pipfile.lock 있을 시에는
pipenv install 로 자동으로 pipefile을 찾아 라이브러리를 설치

# 초기 라이브러리 설치
pipenv install [라이브러리]

# 가상환경 외부에서 파이썬 실행만 가상환경으로 진행
pipenv run [python file]

# 가상환경 진입
pipenv shell
